LEFT JOIN `trns_user_ratings` as `driverrating` ON `trns_requests`.`driver_id`=`driverrating`.`receiver_user_id`
LEFT JOIN `trns_user_ratings` as `requsertating` ON `trns_requests`.`user_id`=`requsertating`.`receiver_user_id` AND `trns_requests`.`request_id`=`requsertating`.`request_id` AND `requsertating`.`is_deleted` ="0"
LEFT JOIN `trns_user_ratings` as `reqtranstating` ON `trns_requests`.`transporter_id`=`reqtranstating`.`receiver_user_id` AND `trns_requests`.`request_id`=`reqtranstating`.`request_id` AND `reqtranstating`.`is_deleted` ="0"
LEFT JOIN `trns_request_bids` as `trans_bid` ON `trns_requests`.`request_id`=`trans_bid`.`request_id` AND `trans_bid`.`user_id` ="50"
WHERE IF(`trns_requests`.`close_bid_time`IS NULL ,  DATE_ADD(NOW(), INTERVAL 2 HOUR)  , `trns_requests`.`close_bid_time`) >= '2021-12-13 15:39:25'
AND `trns_requests`.`request_status` < '2'
AND `trns_requests`.`is_deleted` = 0
GROUP BY `request_id`
ORDER BY `trns_requests`.`pickup_date` DESC, `pickup_time` DESC
